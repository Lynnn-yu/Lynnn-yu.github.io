public class IslandCounter {
    public int numIslands(char[][] grid) {
        if (grid == null || grid.length == 0) return 0;
        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    dfs(grid, i, j); // ËøõË°åDFS
                    count++;
                }
            }
        }
        return count;
    }

    private void dfs(char[][] grid, int x, int y) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y] == '0') {
            return; // Ë∂äÁïåÊàñÊ∞¥ÂüüÔºåËøîÂõû
        }
        grid[x][y] = '0'; // Ê†áËÆ∞‰∏∫Â∑≤ËÆøÈóÆ
        // ÈÄíÂΩíËÆøÈóÆ‰∏ä‰∏ãÂ∑¶Âè≥
        dfs(grid, x + 1, y);
        dfs(grid, x - 1, y);
        dfs(grid, x, y + 1);
        dfs(grid, x, y - 1);
    }
}

import java.util.LinkedList;
import java.util.Queue;

public class IslandCounter {
    public int numIslands(char[][] grid) {
        if (grid == null || grid.length == 0) return 0;
        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    bfs(grid, i, j); // ËøõË°åBFS
                    count++;
                }
            }
        }
        return count;
    }

    private void bfs(char[][] grid, int x, int y) {
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{x, y});
        grid[x][y] = '0'; // Ê†áËÆ∞‰∏∫Â∑≤ËÆøÈóÆ

        int[][] directions = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}}; // Âõõ‰∏™ÊñπÂêë

        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            for (int[] dir : directions) {
                int nextx = cur[0] + dir[0];
                int nexty = cur[1] + dir[1];

                // Ê£ÄÊü•ËæπÁïåÂíåÊ∞¥Âüü
                if (nextx >= 0 && nextx < grid.length && nexty >= 0 && nexty < grid[0].length && grid[nextx][nexty] == '1') {
                    queue.offer(new int[]{nextx, nexty}); // Âä†ÂÖ•ÈòüÂàó
                    grid[nextx][nexty] = '0'; // Ê†áËÆ∞‰∏∫Â∑≤ËÆøÈóÆ
                }
            }
        }
    }
}
